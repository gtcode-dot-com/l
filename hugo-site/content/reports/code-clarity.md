document.addEventListener('DOMContentLoaded', () => {

const smoothScrollLinks = document.querySelectorAll('a[href^="#"]');
smoothScrollLinks.forEach(link => {
link.addEventListener('click', function(e) {
e.preventDefault();
const targetId = this.getAttribute('href');
const targetElement = document.querySelector(targetId);
if (targetElement) {
targetElement.scrollIntoView({ behavior: 'smooth', block: 'start' });
}
});
});

const expertiseSlider = document.getElementById('expertise-slider');
const expertiseLabel = document.getElementById('expertise-label');
const explanation1 = document.getElementById('explanation-1');
const explanation2 = document.getElementById('explanation-2');
const anno1 = document.getElementById('anno-1');
const anno2 = document.getElementById('anno-2');

const explanations = {
novice: [
'<strong><span class="text-green-500">?</span> Operator:</strong> This is Rust\'s "try" operator. If `db::find_user` returns an error, this function will immediately stop and pass that error up. It\'s a shortcut for handling potential failures.',
'<strong><span class="text-green-500">.ok_or()</span> Method:</strong> This converts an `Option` type (which can be something or nothing) into a `Result` type (which can be success or an error). If `user.channel()` returns nothing (`None`), this code creates and returns a `NoChannel` error.'
],
intermediate: [
'<strong><span class="text-green-500">?</span> (Try Operator):</strong> Propagates the `Error` variant of a `Result` from the `db::find_user` call, exiting the function early if it fails.',
'<strong><span class="text-green-500">.ok_or()</span>:</strong> Transforms an `Option` into a `Result`. If the `Option` is `None`, it returns the provided `Error::NoChannel`.'
],
expert: [
'<strong><span class="text-green-500">?</span>:</strong> Propagates error from `db::find_user`.',
'<strong><span class="text-green-500">.ok_or()</span>:</strong> Maps `Option<T>` to `Result<T, E>`, converting `None` to `Err(Error::NoChannel)`.',
]
};

const labels = ['Novice', 'Intermediate', 'Expert'];

function updateExplanations() {
const level = parseInt(expertiseSlider.value);
expertiseLabel.textContent = labels[level];
const key = labels[level].toLowerCase();

explanation1.innerHTML = explanations[key][0];
explanation2.innerHTML = explanations[key][1];

[anno1, anno2, explanation1, explanation2].forEach(el => {
el.style.opacity = 0;
setTimeout(() => el.style.opacity = 1, 50);
});
}

expertiseSlider.addEventListener('input', updateExplanations);
updateExplanations();

const complexitySlider = document.getElementById('complexity');
const impactSlider = document.getElementById('impact');
const magnitudeSlider = document.getElementById('magnitude');
const confidenceSlider = document.getElementById('confidence');

const ctx = document.getElementById('riskChart').getContext('2d');
const riskChart = new Chart(ctx, {
type: 'bar',
data: {
labels: ['Review Priority'],
datasets: [{
label: 'Risk Score',
data: [0],
backgroundColor: ['rgba(2, 132, 199, 0.6)'],
borderColor: ['rgba(2, 132, 199, 1)'],
borderWidth: 1
}]
},
options: {
indexAxis: 'y',
responsive: true,
maintainAspectRatio: false,
scales: {
x: {
beginAtZero: true,
max: 100,
title: {
display: true,
text: 'Calculated Risk Score'
}
}
},
plugins: {
legend: {
display: false
},
tooltip: {
callbacks: {
label: function(context) {
let label = context.dataset.label || '';
if (label) {
label += ': ';
}
if (context.parsed.x !== null) {
label += context.parsed.x.toFixed(2);
}
return label;
}
}
}
}
}
});

function calculateRiskScore() {
const complexity = parseInt(complexitySlider.value);
const impact = parseInt(impactSlider.value);
const magnitude = parseInt(magnitudeSlider.value);
const confidence = parseInt(confidenceSlider.value);

const riskScore = ((complexity * impact * magnitude) / 10000) / (confidence / 10);

const finalScore = Math.min(100, Math.max(0, riskScore));

riskChart.data.datasets[0].data[0] = finalScore;

const scoreColor = finalScore > 75 ? 'rgba(220, 38, 38, 0.6)' : finalScore > 50 ? 'rgba(234, 179, 8, 0.6)' : 'rgba(2, 132, 199, 0.6)';
const borderColor = finalScore > 75 ? 'rgba(220, 38, 38, 1)' : finalScore > 50 ? 'rgba(234, 179, 8, 1)' : 'rgba(2, 132, 199, 1)';

riskChart.data.datasets[0].backgroundColor[0] = scoreColor;
riskChart.data.datasets[0].borderColor[0] = borderColor;

riskChart.update();
}

[complexitySlider, impactSlider, magnitudeSlider, confidenceSlider].forEach(slider => {
slider.addEventListener('input', calculateRiskScore);
});
calculateRiskScore();
});
